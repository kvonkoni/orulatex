\begin{question}
	Solve for $x$ and $y$.
	\begin{exercises}
		\item%
		\begin{eqnarray*}
			x-y & = & 4 \\
			x^2-y^2 & = & 32 \\
		\end{eqnarray*} \answer{$(6,2)$}
		\item%
		\begin{eqnarray*}
			2x^2+3y & = & 10 \\
			x^2-y & = & 8 \\
		\end{eqnarray*} \answer{$(2.61,-1.20)$ and $(-2.61,-1.20)$}
		\item%
		\begin{eqnarray*}
			x-y & = & 11 \\
			xy+28 & = & 0 \\
		\end{eqnarray*} \answer{$(4,-7)$ and $(7,-4)$}
	\end{exercises}
\end{question}
